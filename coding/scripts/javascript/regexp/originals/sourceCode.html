package cod3r;

/*
* Prints the student's grade.
*/

public class grade {
    // entry point of a JAVA program
    public static void main(String[] args) {
        int grade = 7;

        if(grade >= 7) {
            System.out.println("Approved");
        } else {
            System.out.println("Disapproved");
        }
    }

}
